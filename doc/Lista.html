def add(*args)
  args.each do |value|
    if @head == nil then
      @head = Node.new(value, nil, nil)
      @lastnext = @head
    else
      last = Node.new(value, nil, @lastnext)
      @lastnext.next = last
      @lastnext = last
    end
  end
end

def to_a
  a_value = Array.new
  pointer = @head
  while pointer.next != nil do
    a_value.push(pointer.value)

  a_value.push(pointer.value)
